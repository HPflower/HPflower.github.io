# 代码块测试

```C
void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim){
	if(htim == (&htim6)){
		KEY3.Key_Status = HAL_GPIO_ReadPin(GPIOE,GPIO_PIN_4);
		switch(KEY3.judge_Value){
			case 0:
				if(KEY3.Key_Status == 1){
					KEY3.Key_Time = 0;
					KEY3.judge_Value = 0;
				}
				else{
					KEY3.judge_Value = 1;
				}
				break;
			case 1:
				if(KEY3.Key_Status == 1){	
					KEY3.judge_Value = 0;
				}
				else {
					KEY3.judge_Value = 2;
				}
				break;
			case 2:
				if(KEY3.Key_Status == 1){
					KEY3.judge_Value = 0;
					if(KEY3.Key_Time < 200){		//0.5s
						U1_printf_DMA("按钮被单击");
					}
				}
				else{
					KEY3.Key_Time++;
					if(KEY3.Key_Time >= 200){		//2s
						U1_printf_DMA("按钮被长按");
					}
				}
				break;
		}
	}
}
```
